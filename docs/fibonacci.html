def fibonacci(num):
    if num <= 0:
        return "Enter valid positive integer."

    a, b = 1, 1
    print(a, end=' ')

    if num > 1:
        print(b, end=' ')

    for _ in range(2, num):
        a, b = b, a + b
        print(b, end=' ')
