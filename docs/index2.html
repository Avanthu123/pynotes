class Solution:
    def isPalindrome(self, x: int) -> bool:
        pal, new = 0, x
        n = len(str(x))
        if x < 0:
            return False
        else:
            while n > 0:
                pal += (x % 10) * (10 ** (n-1))
                n -= 1
                x //= 10
            if pal == new:
                return True
            else:
                return False
