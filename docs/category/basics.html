from datetime import datetime

def get_age(d):
    d1 = datetime.now()
    months = (d1.year - d.year) * 12 + d1.month - d.month

    year = int(months / 12)
    return year

age = get_age(datetime(1991, 1, 1))

def factorial(num):
    if num < 0:
        return "Enter valid integer."
    fact = 1
    for i in range(1, num+1):
          fact *= i
    return fact

def fibonacci(num):
    if num <= 0:
        return "Enter valid positive integer."

    a, b = 1, 1
    print(a, end=' ')

    if num > 1:
        print(b, end=' ')

    for _ in range(2, num):
        a, b = b, a + b
        print(b, end=' ')
