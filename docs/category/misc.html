from bs4 import BeautifulSoup
import requests

r = requests.get('https://en.wikipedia.org/wiki/2024_Indian_Premier_League')

soup = BeautifulSoup(r.content, 'html.parser')

tags = soup.find_all('h2')
for i in tags:
    print("\nArticle: ", i.text)
