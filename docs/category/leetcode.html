https://leetcode.com/problems/removing-stars-from-a-string/

class Solution:
    def removeStars(self, s: str) -> str:
        nstar = []
        for i in s:
            if i == '*':
                nstar.pop()
            else:
                nstar.append(i)
        return ''.join(nstar)

        arr = nums1+nums2

        n = len(arr)
        if n%2 == 0:
            median = (arr[n//2]+arr[(n//2)-1])/2

class Solution:
    def isPalindrome(self, x: int) -> bool:
        pal, new = 0, x
        n = len(str(x))
        if x < 0:
            return False
        else:
            while n > 0:
                pal += (x % 10) * (10 ** (n-1))
                n -= 1
                x //= 10
            if pal == new:
                return True
            else:
                return False
